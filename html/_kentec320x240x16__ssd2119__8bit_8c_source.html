54 // This driver operates in four different screen orientations. They are:
55 //
56 // * Portrait - The screen is taller than it is wide, and the flex connector is
57 // on the left of the display. This is selected by defining
58 // PORTRAIT.
59 //
60 // * Landscape - The screen is wider than it is tall, and the flex connector is
61 // on the bottom of the display. This is selected by defining
62 // LANDSCAPE.
63 //
64 // * Portrait flip - The screen is taller than it is wide, and the flex
65 // connector is on the right of the display. This is
66 // selected by defining PORTRAIT_FLIP.
67 //
68 // * Landscape flip - The screen is wider than it is tall, and the flex
69 // connector is on the top of the display. This is
70 // selected by defining LANDSCAPE_FLIP.
71 //
72 // These can also be imagined in terms of screen rotation; if portrait mode is
73 // 0 degrees of screen rotation, landscape is 90 degrees of counter-clockwise
74 // rotation, portrait flip is 180 degrees of rotation, and landscape flip is
75 // 270 degress of counter-clockwise rotation.
76 //
77 // If no screen orientation is selected, "landscape flip" mode will be used.
78 //
79 //*****************************************************************************

81  ! defined(LANDSCAPE) && ! defined(LANDSCAPE_FLIP)

83 //#define PORTRAIT
84 #endif

278 //*****************************************************************************
279 //
280 // Writes a data word to the SSD2119. This function implements the basic GPIO
281 // interface to the LCD display.
282 //
283 //*****************************************************************************
284 static void
285 WriteDataGPIO(uint16_t usData)
286 {
287  //
288  // Write the most significant byte of the data to the bus.
289  //
290  SET_LCD_DATA(usData >> 8);
291 
292  HWREG(LCD_CS_BASE + GPIO_O_DATA + (LCD_CS_PIN << 2)) = 0;
293 
294  //
295  // Assert the write enable signal. Do this three times to ensure that we
296  // meet the display timing requirements (20nS per instruction at 50MHz
297  // means we need to ensure at least 3 instructions between edges on WR)
298  //
299  HWREG(LCD_WR_BASE + GPIO_O_DATA + (LCD_WR_PIN << 2)) = 0;
300  HWREG(LCD_WR_BASE + GPIO_O_DATA + (LCD_WR_PIN << 2)) = 0;
301  HWREG(LCD_WR_BASE + GPIO_O_DATA + (LCD_WR_PIN << 2)) = 0;
302 
303  //
304  // Deassert the write enable signal.
305  //
306  HWREG(LCD_WR_BASE + GPIO_O_DATA + (LCD_WR_PIN << 2)) = LCD_WR_PIN;
307 
308  //
309  // Write the least significant byte of the data to the bus.
310  //
311  SET_LCD_DATA(usData);
312 
313  //
314  // Assert the write enable signal.
315  //
316  HWREG(LCD_WR_BASE + GPIO_O_DATA + (LCD_WR_PIN << 2)) = 0;
317  HWREG(LCD_WR_BASE + GPIO_O_DATA + (LCD_WR_PIN << 2)) = 0;
318  HWREG(LCD_WR_BASE + GPIO_O_DATA + (LCD_WR_PIN << 2)) = 0;
319 
320  //
321  // Deassert the write enable signal.
322  //
323  HWREG(LCD_WR_BASE + GPIO_O_DATA + (LCD_WR_PIN << 2)) = LCD_WR_PIN;
324  HWREG(LCD_WR_BASE + GPIO_O_DATA + (LCD_WR_PIN << 2)) = LCD_WR_PIN;
325  HWREG(LCD_WR_BASE + GPIO_O_DATA + (LCD_WR_PIN << 2)) = LCD_WR_PIN;
326 
327  HWREG(LCD_CS_BASE + GPIO_O_DATA + (LCD_CS_PIN << 2)) = LCD_CS_PIN;
328 
329 }

331 //*****************************************************************************
332 //
333 // Writes a command to the SSD2119. This function implements the basic GPIO
334 // interface to the LCD display.
335 //
336 //*****************************************************************************
337 static void
338 WriteCommandGPIO(uint8_t ucData)
339 
340 {
341  //
342  // Write the most significant byte of the data to the bus. This is always
343  // 0 since commands are no more than 8 bits currently.
344  //
345  SET_LCD_DATA(0);
346 
347  HWREG(LCD_CS_BASE + GPIO_O_DATA + (LCD_CS_PIN << 2)) = 0;
348 
349  //
350  // Assert the write enable and DC signals. Do this 3 times to slow things
351  // down a bit.
352  //
353  HWREG(LCD_WR_BASE + GPIO_O_DATA + ((LCD_WR_PIN | LCD_DC_PIN) << 2)) = 0;
354  HWREG(LCD_WR_BASE + GPIO_O_DATA + ((LCD_WR_PIN | LCD_DC_PIN) << 2)) = 0;
355  HWREG(LCD_WR_BASE + GPIO_O_DATA + ((LCD_WR_PIN | LCD_DC_PIN) << 2)) = 0;
356 
357  //
358  // Deassert the write enable and DC signals. We need to leave WR high for
359  // at least 50nS so, again, stick in dummy writes to pad the timing.
360  //
361  HWREG(LCD_WR_BASE + GPIO_O_DATA + ((LCD_WR_PIN | LCD_DC_PIN) << 2)) =
362  (LCD_WR_PIN | LCD_DC_PIN);
363  HWREG(LCD_WR_BASE + GPIO_O_DATA + ((LCD_WR_PIN | LCD_DC_PIN) << 2)) =
364  (LCD_WR_PIN | LCD_DC_PIN);
365  HWREG(LCD_WR_BASE + GPIO_O_DATA + ((LCD_WR_PIN | LCD_DC_PIN) << 2)) =
366  (LCD_WR_PIN | LCD_DC_PIN);
367 
368  //
369  // Write the least significant byte of the data to the bus.
370  //
371  SET_LCD_DATA(ucData);
372 
373  //
374  // Assert the write enable signal. Again, do this three times to ensure
375  // we meet the timing spec.
376  //
377  HWREG(LCD_WR_BASE + GPIO_O_DATA + ((LCD_WR_PIN | LCD_DC_PIN) << 2)) = 0;
378  HWREG(LCD_WR_BASE + GPIO_O_DATA + ((LCD_WR_PIN | LCD_DC_PIN) << 2)) = 0;
379  HWREG(LCD_WR_BASE + GPIO_O_DATA + ((LCD_WR_PIN | LCD_DC_PIN) << 2)) = 0;
380 
381  //
382  // Deassert the write enable and DC signals. Make sure we add padding here
383  // too since some compilers inline this function.
384  //
385  HWREG(LCD_WR_BASE + GPIO_O_DATA + ((LCD_WR_PIN | LCD_DC_PIN) << 2)) =
386  (LCD_WR_PIN | LCD_DC_PIN);
387  HWREG(LCD_WR_BASE + GPIO_O_DATA + ((LCD_WR_PIN | LCD_DC_PIN) << 2)) =
388  (LCD_WR_PIN | LCD_DC_PIN);
389  HWREG(LCD_WR_BASE + GPIO_O_DATA + ((LCD_WR_PIN | LCD_DC_PIN) << 2)) =
390  (LCD_WR_PIN | LCD_DC_PIN);
391 
392  HWREG(LCD_CS_BASE + GPIO_O_DATA + (LCD_CS_PIN << 2)) = LCD_CS_PIN;
393 }

473 void
474 Kentec320x240x16_SSD2119Init(void)
475 {
476  uint32_t ulClockMS, ulCount;
477 
478  //
479  // Get the current processor clock frequency.
480  //
481  ulClockMS = ROM_SysCtlClockGet() / (3 * 1000);
482 
483  //
484  // Enable the GPIO peripherals used to interface to the SSD2119.
485  //
486 
487 
488 
489  ROM_SysCtlPeripheralEnable(LCD_DATAH_PERIPH);
490  ROM_SysCtlPeripheralEnable(LCD_DC_PERIPH);
491  ROM_SysCtlPeripheralEnable(LCD_RD_PERIPH);
492  ROM_SysCtlPeripheralEnable(LCD_WR_PERIPH);
493  // ROM_SysCtlPeripheralEnable(LCD_RST_PERIPH);
494  ROM_SysCtlPeripheralEnable(LCD_CS_PERIPH);
495  ROM_SysCtlPeripheralEnable(LCD_BACKLIGHT_PERIPH);
496 
497  //
498  // Perform low level interface initialization depending upon how the LCD
499  // is connected to the Stellaris microcontroller. This varies depending
500  // upon the daughter board connected it is possible that a daughter board
501  // can drive the LCD directly rather than via the basic GPIO interface.
502  //
503  {
504  //
505  // Initialize the GPIOs used to interface to the LCD controller.
506  //
507  InitGPIOLCDInterface(ulClockMS);
508  }
509 
510  //
511  // Enter sleep mode (if we are not already there).
512  //
513  WriteCommand(SSD2119_SLEEP_MODE_REG);
514  WriteData(0x0001);
515 
516  //
517  // Set initial power parameters.
518  //
519  WriteCommand(SSD2119_PWR_CTRL_5_REG);
520  WriteData(0x00BA);
521  WriteCommand(SSD2119_VCOM_OTP_1_REG);
522  WriteData(0x0006);
523 
524  //
525  // Start the oscillator.
526  //
527  WriteCommand(SSD2119_OSC_START_REG);
528  WriteData(0x0001);
529 
530  //
531  // Set pixel format and basic display orientation (scanning direction).
532  //
533  WriteCommand(SSD2119_OUTPUT_CTRL_REG);
534  WriteData(0x30EF);
535  WriteCommand(SSD2119_LCD_DRIVE_AC_CTRL_REG);
536  WriteData(0x0600);
537 
538  //
539  // Exit sleep mode.
540  //
541  WriteCommand(SSD2119_SLEEP_MODE_REG);
542  WriteData(0x0000);
543 
544  //
545  // Delay 30mS
546  //
547  ROM_SysCtlDelay(30 * ulClockMS);
548  //ROM_SysCtlDelay(ROM_SysCtlClockGet() / (4 * 30));
549 
550  //
551  // Configure pixel color format and MCU interface parameters.
552  //
553  WriteCommand(SSD2119_ENTRY_MODE_REG);
554  WriteData(ENTRY_MODE_DEFAULT);
555 
556  //
557  // Enable the display.
558  //
559  WriteCommand(SSD2119_DISPLAY_CTRL_REG);
560  WriteData(0x0033);
561 
562  //
563  // Set VCIX2 voltage to 6.1V.
564  //
565  WriteCommand(SSD2119_PWR_CTRL_2_REG);
566  WriteData(0x0005);
567 
568  //
569  // Configure gamma correction.
570  //
571  WriteCommand(SSD2119_GAMMA_CTRL_1_REG);
572  WriteData(0x0000);
573  WriteCommand(SSD2119_GAMMA_CTRL_2_REG);
574  WriteData(0x0400);
575  WriteCommand(SSD2119_GAMMA_CTRL_3_REG);
576  WriteData(0x0106);
577  WriteCommand(SSD2119_GAMMA_CTRL_4_REG);
578  WriteData(0x0700);
579  WriteCommand(SSD2119_GAMMA_CTRL_5_REG);
580  WriteData(0x0002);
581  WriteCommand(SSD2119_GAMMA_CTRL_6_REG);
582  WriteData(0x0702);
583  WriteCommand(SSD2119_GAMMA_CTRL_7_REG);
584  WriteData(0x0707);
585  WriteCommand(SSD2119_GAMMA_CTRL_8_REG);
586  WriteData(0x0203);
587  WriteCommand(SSD2119_GAMMA_CTRL_9_REG);
588  WriteData(0x1400);
589  WriteCommand(SSD2119_GAMMA_CTRL_10_REG);
590  WriteData(0x0F03);
591 
592  //
593  // Configure Vlcd63 and VCOMl.
594  //
595  WriteCommand(SSD2119_PWR_CTRL_3_REG);
596  WriteData(0x0007);
597  WriteCommand(SSD2119_PWR_CTRL_4_REG);
598  WriteData(0x3100);
599 
600  //
601  // Set the display size and ensure that the GRAM window is set to allow
602  // access to the full display buffer.
603  //
604  WriteCommand(SSD2119_V_RAM_POS_REG);
605  WriteData((LCD_VERTICAL_MAX-1) << 8);

691 static void
692 Kentec320x240x16_SSD2119PixelDrawMultiple(void *pvDisplayData, int32_t lX,
693  int32_t lY, int32_t lX0, int32_t lCount,
694  int32_t lBPP,
695  const uint8_t *pucData,
696  const uint8_t *pucPalette)
697 {
698  uint32_t ulByte;
699 
700  //
701  // Set the cursor increment to left to right, followed by top to bottom.
702  //
703  WriteCommand(SSD2119_ENTRY_MODE_REG);
704  WriteData(MAKE_ENTRY_MODE(HORIZ_DIRECTION));
705 
706  //
707  // Set the starting X address of the display cursor.
708  //
709  WriteCommand(SSD2119_X_RAM_ADDR_REG);
710  WriteData(MAPPED_X(lX, lY));
711 
712  //
713  // Set the Y address of the display cursor.
714  //
715  WriteCommand(SSD2119_Y_RAM_ADDR_REG);
716  WriteData(MAPPED_Y(lX, lY));
717 
718  //
719  // Write the data RAM write command.
720  //
721  WriteCommand(SSD2119_RAM_DATA_REG);
722 
723  //
724  // Determine how to interpret the pixel data based on the number of bits
725  // per pixel.
726  //
727  switch(lBPP)
728  {
729  //
730  // The pixel data is in 1 bit per pixel format.
731  //
732  case 1:
733  {
734  //
735  // Loop while there are more pixels to draw.
736  //
737  while(lCount)
738  {
739  //
740  // Get the next byte of image data.
741  //
742  ulByte = *pucData++;
743 
744  //
745  // Loop through the pixels in this byte of image data.
746  //
747  for(; (lX0 < 8) && lCount; lX0++, lCount--)
748  {
749  //
750  // Draw this pixel in the appropriate color.
751  //
752  WriteData(((uint32_t *)pucPalette)[(ulByte >>
753  (7 - lX0)) & 1]);
754  }
755 
756  //
757  // Start at the beginning of the next byte of image data.
758  //
759  lX0 = 0;
760  }
761 
762  //
763  // The image data has been drawn.
764  //
765  break;
766  }
767 
768  //
769  // The pixel data is in 4 bit per pixel format.
770  //
771  case 4:
772  {
773  //
774  // Loop while there are more pixels to draw. "Duff's device" is
775  // used to jump into the middle of the loop if the first nibble of
776  // the pixel data should not be used. Duff's device makes use of
777  // the fact that a case statement is legal anywhere within a
778  // sub-block of a switch statement. See
779  // http://en.wikipedia.org/wiki/Duff's_device for detailed
780  // information about Duff's device.
781  //
782  switch(lX0 & 1)
783  {
784  case 0:
785  while(lCount)
786  {
787  //
788  // Get the upper nibble of the next byte of pixel data
789  // and extract the corresponding entry from the
790  // palette.
791  //
792  ulByte = (*pucData >> 4) * 3;
793  ulByte = (*(uint32_t *)(pucPalette + ulByte) &
794  0x00ffffff);
795 
796  //
797  // Translate this palette entry and write it to the
798  // screen.
799  //
800  WriteData(DPYCOLORTRANSLATE(ulByte));
801 
802  //
803  // Decrement the count of pixels to draw.
804  //
805  lCount--;
806 
807  //
808  // See if there is another pixel to draw.
809  //
810  if(lCount)
811  {
812  case 1:
813  //
814  // Get the lower nibble of the next byte of pixel
815  // data and extract the corresponding entry from
816  // the palette.
817  //
818  ulByte = (*pucData++ & 15) * 3;
819  ulByte = (*(uint32_t *)(pucPalette + ulByte) &
820  0x00ffffff);
821 
822  //
823  // Translate this palette entry and write it to the
824  // screen.
825  //
826  WriteData(DPYCOLORTRANSLATE(ulByte));
827 
828  //
829  // Decrement the count of pixels to draw.
830  //
831  lCount--;
832  }
833  }
834  }
835 
836  //
837  // The image data has been drawn.
838  //
839  break;
840  }
841 
842  //
843  // The pixel data is in 8 bit per pixel format.
844  //
845  case 8:
846  {
847  //
848  // Loop while there are more pixels to draw.
849  //
850  while(lCount--)
851  {
852  //
853  // Get the next byte of pixel data and extract the
854  // corresponding entry from the palette.
855  //
856  ulByte = *pucData++ * 3;
857  ulByte = *(uint32_t *)(pucPalette + ulByte) & 0x00ffffff;
858 
859  //
860  // Translate this palette entry and write it to the screen.
861  //
862  WriteData(DPYCOLORTRANSLATE(ulByte));
863  }
864 
865  //
866  // The image data has been drawn.
867  //
868  break;
869  }
870 
871  //
872  // We are being passed data in the display's native format. Merely
873  // write it directly to the display. This is a special case which is
874  // not used by the graphics library but which is helpful to
875  // applications which may want to handle, for example, JPEG images.
876  //
877  case 16:
878  {
879  uint16_t usByte;
880 
881  //
882  // Loop while there are more pixels to draw.
883  //
884  while(lCount--)
885  {
886  //
887  // Get the next byte of pixel data and extract the
888  // corresponding entry from the palette.
889  //
890  usByte = *((uint16_t *)pucData);
891  pucData += 2;
892 
893  //
894  // Translate this palette entry and write it to the screen.
895  //
896  WriteData(usByte);
897  }
898  }
899  }
